async function fetchVenues(la1, ln1, la2, ln2) { // Calculate Midpoint const mLat = (parseFloat(la1) + parseFloat(la2)) / 2; const mLng = (parseFloat(ln1) + parseFloat(ln2)) / 2; const list = document.getElementById('venue-list'); list.innerHTML = "Consulting the Oracle..."; try { // Invoke the Supabase Edge Function const { data, error } = await _supabase.functions.invoke('oracle-search', { body: { lat: mLat, lng: mLng } }); if (error) throw error; list.innerHTML = ''; // Clear loading text if (data.results && data.results.length > 0) { data.results.forEach(v => { const card = document.createElement('div'); card.className = 'venue-card'; // Display Name and Formatted Address card.innerHTML = `${v.name.toUpperCase()}
${v.location.formatted_address || "Sharjah Sector"}`; list.appendChild(card); }); } else { list.innerHTML = "[ ORACLE SILENT ]: NO POIs RECORDED IN THIS SECTOR."; } } catch (e) { console.error("ORACLE_SYNC_ERROR:", e); list.innerHTML = `CONNECTION FAILED: ${e.message}`; } document.getElementById('oracle-ballot').style.display = 'block'; }
